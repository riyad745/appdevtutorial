Introduction to Dart

Dart is a client-optimized language developed by Google. It compiles to native ARM code for mobile/desktop and to JavaScript for the web. It's strongly typed, object-oriented, and powers all Flutter apps.

üîí Type Safe

Dart's sound type system catches errors at compile time, not runtime.

‚ö° Fast

Compiles AOT (Ahead of Time) to native machine code for maximum performance.

üåê Versatile

Runs on mobile, desktop, web (via JS compilation), and server.

üê¶ Flutter-Powered

The exclusive language of Flutter ‚Äî learn Dart, unlock Flutter.

Null Safety ‚≠ê

Dart's null safety (introduced in Dart 2.12) eliminates entire classes of null-pointer bugs. By default, variables cannot be null ‚Äî you must explicitly opt in with ?. 

// Non-nullable (default) ‚Äî cannot be null
String name = 'Alice';
// name = null; ‚Üê COMPILE ERROR

// Nullable ‚Äî add ? to allow null
String? nickname = null;  // OK

// Null-aware operators
print(nickname ?? 'No nickname'); // ?? = use right if left is null
nickname ??= 'Default';           // assign only if null
print(nickname?.length);          // ?. = call only if not null
print(nickname!.length);          // ! = assert not null (throws if null)

// late ‚Äî non-nullable but initialized later
late String data;
data = 'assigned later'; // must assign before use or throws
‚ö†Ô∏è Avoid ! (bang operator) unless you're absolutely certain the value isn't null. Prefer ??, ?., or null checks.

Control Flow

// if / else
if (age >= 18) { print('Adult'); } else { print('Minor'); }

// for loop
for (int i = 0; i < 5; i++) print(i);

// for-in loop
for (var item in ['a', 'b', 'c']) print(item);

// while / do-while
while (count > 0) count--;
do { print(n); n--; } while (n > 0);

// switch (Dart 3+ supports pattern matching!)
switch (color) {
  case 'red':   print('Stop'); break;
  case 'green': print('Go'); break;
  default:      print('Wait');
}

// Dart 3+ switch expressions
var label = switch (color) {
  'red'   => 'Stop',
  'green' => 'Go',
  _       => 'Wait',   // _ = default wildcard
};

OOP & Classes

class Animal {
  final String name;
  int _age; // _ prefix = private

  // Constructor
  Animal(this.name, this._age); // shorthand this.

  // Named constructor
  Animal.unknown() : name = 'Unknown', _age = 0;

  // Getter / Setter
  int get age => _age;
  set age(int v) { if (v >= 0) _age = v; }

  void speak() => print('$name makes a sound');

  // Static method
  static void describe() => print('Animals are living beings');
}

class Dog extends Animal {
  String breed;
  Dog(String name, this.breed) : super(name, 0);

  @override
  void speak() => print('$name barks! üê∂');
}

abstract class Shape {
  double area();          // must be implemented
  void describe() => print('Area: ${area()}');
}

class Circle implements Shape {
  final double radius;
  Circle(this.radius);
  @override double area() => 3.14 * radius * radius;
}

Mixins & Extensions

// Mixin ‚Äî reuse code across class hierarchies
mixin Swimmable {
  void swim() => print('Swimming!');
}
mixin Flyable {
  void fly() => print('Flying!');
}

class Duck extends Animal with Swimmable, Flyable {
  Duck() : super('Duck', 2);
}
var duck = Duck();
duck.swim(); duck.fly(); duck.speak();

// Extension ‚Äî add methods to existing types
extension StringUtils on String {
  String capitalize() =>
    isEmpty ? this : this[0].toUpperCase() + substring(1);
  bool get isEmail => contains('@');
}

print('hello'.capitalize()); // 'Hello'
print('a@b.com'.isEmail);    // true

Async & Futures

// Future ‚Äî a value that will be available later
Future<String> fetchData() async {
  await Future.delayed(Duration(seconds: 2));
  return 'Data loaded!';
}

void main() async {
  print('Fetching...');
  try {
    var data = await fetchData();
    print(data); // 'Data loaded!'
  } catch (e) {
    print('Error: $e');
  }
}

// Future combinators
await Future.wait([fetchA(), fetchB()]); // parallel
await Future.any([fast(), slow()]);       // first to complete

// HTTP request with dart:io / http package
import 'package:http/http.dart' as http;

var url = Uri.parse('https://api.example.com/users');
var res = await http.get(url);
print(res.body);

Streams

A Stream is like a Future but delivers multiple values over time ‚Äî perfect for live data, events, or large datasets. 

// Creating a stream
Stream<int> counter(int max) async* {
  for (int i = 0; i <= max; i++) {
    await Future.delayed(Duration(seconds: 1));
    yield i;  // emit value
  }
}

// Consuming a stream
await for (var n in counter(5)) {
  print(n); // 0, 1, 2, 3, 4, 5 (one per second)
}

// Stream methods
counter(10)
  .where((n) => n % 2 == 0)   // filter evens
  .map((n) => n * n)            // square them
  .listen((v) => print(v));    // subscribe

// StreamController ‚Äî create streams manually
var controller = StreamController<String>();
controller.stream.listen(print);
controller.sink.add('event 1');
controller.close();

Error Handling

// try / catch / finally
try {
  var n = int.parse('not a number');
} on FormatException catch (e) {
  print('Format error: $e'); // specific error type
} catch (e, stackTrace) {
  print('Unknown error: $e');  // catch-all
} finally {
  print('Always runs');
}

// Custom exceptions
class AuthException implements Exception {
  final String message;
  AuthException(this.message);
  @override String toString() => 'AuthException: $message';
}

throw AuthException('Token expired');
